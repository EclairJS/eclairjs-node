var Utils = require('../../utils.js');

var gKernelP;

/**
 * @constructor
 * @classdesc A field inside a StructType.
 * @memberof module:eclairjs/sql/types
 * @param {string} name - The name of this field.
 * @param {DataType} dataType - The data type of this field.
 * @param {boolean}  nullable -  Indicates if values of this field can be null values.
 * @param {Metadata} metadata - The metadata of this field. The metadata should be preserved during transformation if the content of the column is not modified, e.g, in selection.
 */
function StructField() {
  if (arguments.length == 2) {
    // Someone created an instance of this class for us
    this.kernelP = arguments[0];
    this.refIdP = arguments[1];
  } else {
    this.kernelP = gKernelP;

    var args = {
      target: StructField,
      args: Utils.wrapArguments(arguments),
      kernelP: gKernelP
    };

    this.refIdP = Utils.generateConstructor(args);
  }
}

/**
 * @returns {string}
 */
StructField.prototype.name = function() {
  var args = {
    target: this,
    method: 'name',
    returnType: String
  };

  return Utils.generate(args);
};

StructField.moduleLocation = '/sql/types/StructField';

module.exports = function(kP) {
  gKernelP = kP;

  return StructField;
};
